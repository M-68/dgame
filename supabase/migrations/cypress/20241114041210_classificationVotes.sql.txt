create table
  public.votes (
    id bigint generated by default as identity,
    user_id uuid not null,
    classification_id bigint null,
    anomaly_id bigint null,
    created_at timestamp with time zone not null default now(),
    upload bigint null,
    constraint votes_pkey primary key (id),
    constraint votes_user_id_classification_id_key unique (user_id, classification_id),
    constraint votes_user_id_fkey foreign key (user_id) references profiles (id),
    constraint votes_classification_id_fkey foreign key (classification_id) references classifications (id),
    constraint votes_anomaly_id_fkey foreign key (anomaly_id) references anomalies (id),
    constraint votes_upload_fkey foreign key (upload) references uploads (id)
  ) tablespace pg_default;
